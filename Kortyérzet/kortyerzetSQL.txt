DROP TABLE IF EXISTS checkin;
DROP TABLE IF EXISTS beer;
DROP TABLE IF EXISTS brewery;
DROP TABLE IF EXISTS users;


CREATE TABLE users(
    user_id SERIAL PRIMARY KEY,
    user_name VARCHAR(50) NOT NULL,
    user_password VARCHAR(100) NOT NULL,
    user_email VARCHAR(50) NOT NULL,
    user_role VARCHAR(6) NOT NULL
);

CREATE TABLE brewery (
	brewery_ID SERIAL PRIMARY KEY,
	brewery_name TEXT,
	brewery_logo TEXT,
	brewery_HQ TEXT,
	brewery_desc TEXT,
	brewery_beerCount INT,
	brewery_rating INT,
	brewery_timesRated INT,
	brewery_checkin INT
);
CREATE TABLE beer (
	beer_ID SERIAL PRIMARY KEY,
	beer_name TEXT,
	beer_logo TEXT,
	beer_style TEXT,
	beer_desc TEXT,
	beer_rating INT,
	beer_timesRated INT,
	beer_checkin INT,
	beer_abv DECIMAL(3,1),
	beer_ibu INT,
    brewery_ID INT REFERENCES brewery(brewery_ID) on DELETE CASCADE

);

CREATE TABLE checkin (
	user_ID  INT REFERENCES users(user_id) on DELETE CASCADE,
	beer_ID  INT REFERENCES beer(beer_ID) on DELETE CASCADE,
	checkin_comment TEXT,
	checkin_rating DECIMAL(3,2),
	checkin_img TEXT

);
